% Her definerer jeg min "makro" som laver en udregning for mig
\newenvironment{faktiskspænding}[3] %{Spænding}{spændingsfald}{ navn startspænding | navn slutspænding | fase eller netspænding}
{
% I følgende udskifter jeg kommaer med punktum i mine variabler. FP-pakken regner med punktum.
% Hver variabel starter med # og derefter nummeret i rækken.
\setsepchar{ | }
\readlist\Nam{#3}
\StrSubstitute{#1}{,}{.}[\Ustart]%
\StrSubstitute{#2}{,}{.}[\dU]%

%Udregn strømmen
\FPeval\slut{round(\Ustart * (1 - \dU):2)}

\begin{align*}
	U_{ \Nam[2] } &= U_{ \Nam[1] } \cdot \left( 1 - \Delta U_{ \Nam[3] } \right) \\
				&= \Ustart \cdot (1-\dU) \\
				&= \slut \hspace{2pt} V
\end{align*}
}
