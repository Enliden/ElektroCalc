\newenvironment{tid-inversrelæ}[6]
%{K} {C} {I} {I-hak} {a } { tidnavn | strømnavn }
{
\setsepchar{ | }
\readlist\Nam{#6}
%Her udskiftes komma med punktum
\StrSubstitute{#1}{,}{.}[\K]%
\StrSubstitute{#2}{,}{.}[\C]%
\StrSubstitute{#3}{,}{.}[\I]%
\StrSubstitute{#1}{,}{.}[\Ihak]%
\StrSubstitute{#2}{,}{.}[\a]%


%Da kvadratrod 3 bruges ofte defineres den her.
\FPeval\sqTre{root(2,3)}

%Udregn tiden
\FPeval\tInvers{\K * \C /( pow(\a,\I / \Ihak) - 1 )}


\begin{align*}
t_{\Nam[1]} &= K \cdot \frac{C}{ \left( \frac{I}{I_{hak}}\right)^a}   \\
			&= \K \cdot \frac{\C}{ \left( \frac{\I}{\Ihak}\right)^\a} \\
			&= \tInvers
\end{align*}
}
