% Her definerer jeg min "makro" som laver en udregning for mig
\newenvironment{HV-ZtilIk2f}[3]
%{Un}{ R1 | X1 | R2 | X2 | R3 | X3 | R4 | X4 | R5 | X5 }{Ik,navn | Z1 navn | Z2 navn | Z3 navn | Z4 navn | Z5 navn}
{
\setsepchar{ | }
\readlist\Zv{#2}
\readlist\Nam{#3}
%Her udskiftes komma med punktum
\StrSubstitute{#1}{,}{.}[\Un]%
\StrSubstitute{\Zv[1]}{,}{.}[\Ret]%
\StrSubstitute{\Zv[2]}{,}{.}[\Xet]%
\StrSubstitute{\Zv[3]}{,}{.}[\Rto]%
\StrSubstitute{\Zv[4]}{,}{.}[\Xto]%
\StrSubstitute{\Zv[5]}{,}{.}[\Rtre]%
\StrSubstitute{\Zv[6]}{,}{.}[\Xtre]%
\StrSubstitute{\Zv[7]}{,}{.}[\Rfi]%
\StrSubstitute{\Zv[8]}{,}{.}[\Xfi]%
\StrSubstitute{\Zv[9]}{,}{.}[\Rfe]%
\StrSubstitute{\Zv[10]}{,}{.}[\Xfe]%
%Da kvadratrod 3 bruges ofte defineres den her.
\FPeval\sqTre{root(2,3)}

%Udregn samlede R og X
\FPeval\Rtot{\Ret + \Rto + \Rtre + \Rfi + \Rfe}
\FPeval\Xtot{\Xet + \Xto + \Xtre + \Xfi + \Xfe}

%udregn samlede net og trafo impedans
\FPeval\Ztot{root(2,pow(2,\Rtot)+pow(2,\Xtot))}

%udregn kortslutningen 1f
\FPeval\IktoF{round(\Un / (2 * \Ztot):2)}

\FPeval\yepET{1}
\FPeval\yepTO{1}
\FPeval\yepTRE{1}
\FPeval\yepFI{1}
\FPeval\yepFE{1}

\FPeval\Zettest{\Ret + \Xet}
\FPifzero{\Zettest}
\begin{align*}
 Ingen impedanser\\
\end{align*}
\else
\FPeval\yepET{0}
\fi

\FPeval\Ztotest{\Rto + \Xto + \yepET}
\FPifzero{\Ztotest}
\begin{align*}
Ik_{\Nam[1]} &= \frac{U_n}{2 * Z_{\Nam[2]}} \\
	&= \frac{\Un}{2 \cdot (\Ret + j\Xet)}\\
	&= \IktoF \unit{A}
\end{align*}
\else
\FPeval\yepTO{0}
\fi

\FPeval\Ztretest{\Rtre + \Xtre + \yepET + \yepTO}
\FPifzero{\Ztretest}
\begin{align*}
Ik_{\Nam[1]} &= \frac{U_n}{2 * (Z_{\Nam[2]} + Z_{\Nam[3]})} \\
	&= \frac{\Un}{2 \cdot ((\Ret + j\Xet) + (\Rto + j\Xto))}\\
	&= \IktoF \unit{A}
\end{align*}
\else
\FPeval\yepTRE{0}
\fi

\FPeval\Ztretest{\Rfi + \Xfi + \yepET + \yepTO + \yepTRE}
\FPifzero{\Ztretest}
\begin{align*}
Ik_{\Nam[1]} &= \frac{U_n}{2 * (Z_{\Nam[2]} + Z_{\Nam[3]} + Z_{\Nam[4]})} \\
	&= \frac{\Un}{2 \cdot ((\Ret + j\Xet) + (\Rto + j\Xto) + (\Rtre + j\Xtre))}\\
	&= \IktoF \unit{A}
\end{align*}
\else
\FPeval\yepFI{0}
\fi

\FPeval\Ztretest{\Rfe + \Xfe + \yepET + \yepTO + \yepTRE + \yepFI}
\FPifzero{\Ztretest}
\begin{align*}
Ik_{\Nam[1]} &= \frac{U_n}{2 * (Z_{\Nam[2]} + Z_{\Nam[3]} + Z_{\Nam[4]} + Z_{\Nam[5]})} \\
	&= \frac{\Un}{2 \cdot ((\Ret + j\Xet) + (\Rto + j\Xto) + (\Rtre + j\Xtre) + (\Rfi + j\Xfi))}\\
	&= \IktoF \unit{A}
\end{align*}
\else
\FPeval\yepFE{0}
\fi

\FPeval\all{\yepET + \yepTO + \yepTRE + \yepFI + \yepFE}
\FPifzero{\all}
\begin{align*}
Ik_{\Nam[1]} &= \frac{U_n}{2 * (Z_{\Nam[2]} + Z_{\Nam[3]} + Z_{\Nam[4]} + Z_{\Nam[5]} + Z_{\Nam[6]})} \\
	&= \frac{\Un}{2 \cdot ((\Ret + j\Xet) + (\Rto + j\Xto) + (\Rtre + j\Xtre) + (\Rfi + j\Xfi) + (\Rfe + j\Xfe))}\\
	&= \IktoF \unit{A}
\end{align*}
\else
\fi
}
